drawdino proc near 
    mov cx , 120
    mov dx , 170
    
    ;calculate maximum row and column value and store it in the stack 
    add cx , [dWidth]
    add dx , [dheight]
    push dx
    push cx 
   

    drawdinol1: ;drawdino loop 1 
        mov ah , 0ch 
        mov al , [dcolor]
        mov bh , 00h 
        int 10h
        inc cx
        pop bx ; pop the maximum column value form the stack for comparision
        cmp cx , bx 
        push bx ; put the max col value back to the stack 
        jb drawdinol1
        
        mov cx , [dColum] ; reset the column pixel 
        inc dx 

        pop bx 
        pop ax ; pop the maximum row value form the stack for comparision

        cmp dx , ax
        push ax 
        push bx
        jb drawdinol1
 
ret
drawdino endp
